void print_string(char * str)
{
	char * ptr;
	printf("                \"");
	for (ptr = str; *ptr != 0; ptr++){
		if (*ptr == '\\')
			printf("\\\\");
		else if (*ptr == '"')
			printf("\\\"");
		else if (*ptr == '\t')
			printf("\\t");
		else
			putchar(*ptr);
	}
	printf("\",\n");
}
